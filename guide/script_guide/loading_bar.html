$plugins.uiLoading




Methods

Attributes
Name Description parameter
uiLoading() 해당되는 타입, 아이디로 로딩바 처리 id, type, txt, visible 

            // 기본 로딩바 - id가 없을땐 'body'에서 생성
            // - txt가 없을떈 ''컨텐츠 가져오는 중 기본글
            $plugins.uiLoading({ visible: true });
            // or
            $plugins.uiLoading({ type: 'base',visible: true });

            // 닷 로딩바 - id가 없을땐 'body'에서 생성
            // - txt가 없을떈 ''컨텐츠 가져오는 중 기본글
            $plugins.uiLoading({ type: 'dot',visible: true });

            // 글자변환 닷로딩바(퍼블용 샘플입니다)  - id가 없을땐 'body'에서 생성
            // - txt가 없을떈 '
              고객님이 입력한 정보를

              확인하고 있어요
            
            삼성화재는 부수거래조건이 없어요
            
              삼성화재는 시중은행 대비

              10% 높은 DSR을 적용해드리니

              높은한도를 비교, 확인해보세요
            ' 기본글
            $plugins.uiLoading({ type: 'dot',visible: true ,txtchange: true});
            
            // 프로그래스 로딩바  - id가 없을땐 'main>.content'에서 생성
            // - txt가 없을떈 '고객님의 계약을 불러오고 있어요.' 기본글
            $plugins.uiLoading({ type: 'progress',visible: true });

            // 로딩바 닫힘
            $plugins.uiLoading({ visible: false });

Attributes
